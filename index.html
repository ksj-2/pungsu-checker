// api/proxy.js export default async function handler(request, response) { // 1. 웹사이트에서 보낸 '주소' 정보를 꺼냅니다. const { address } = request.query; if (!address) { return response.status(400).json({ error: '주소를 입력해주세요.' }); } // 2. Vercel에 숨겨둔 나의 비밀 카카오 API 키를 가져옵니다. const KAKAO_API_KEY = process.env.KAKAO_API_KEY; // 3. 이 키를 가지고 진짜 카카오 서버에 데이터를 요청합니다. const kakaoResponse = await fetch( `https://dapi.kakao.com/v2/local/search/address.json?query=${encodeURIComponent(address)}`, { headers: { Authorization: `KakaoAK ${KAKAO_API_KEY}`, }, } ); const data = await kakaoResponse.json(); // 4. 카카오에서 받은 결과를 웹사이트에 다시 전달해줍니다. // CORS 헤더 추가: 다른 도메인(GitHub Pages)에서 내 Vercel 서버로 요청할 수 있게 허용해주는 설정 response.setHeader('Access-Control-Allow-Origin', '*'); response.setHeader('Access-Control-Allow-Methods', 'GET, OPTIONS'); response.setHeader('Access-Control-Allow-Headers', 'Content-Type'); return response.status(200).json(data); }
